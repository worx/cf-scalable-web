Here you go — a working Drupal 11 vhost. I'll write it both as a here-doc you can paste on the deploy-host AND propose where it should live in the repo so it's version-controlled long term.

**The vhost** (`/var/www/nginx/sites-enabled/drupal.conf` on FSx, which appears as `/etc/nginx/shared/sites-enabled/drupal.conf` on nginx boxes):

```bash
sudo tee /var/www/nginx/sites-enabled/drupal.conf > /dev/null <<'NGXEOF'
# SPDX-License-Identifier: GPL-2.0-or-later
# Copyright (C) 2026 The Worx Company
# Author: Kurt Vanderwater <kurt@worxco.net>
#
# Drupal 11 vhost for cf-scalable-drupal sandbox.
# Lives on FSx at /fsx/nginx/sites-enabled/drupal.conf, picked up by every
# nginx instance via the NFS mount to /etc/nginx/shared/sites-enabled/.
# Edit-in-place: changes take effect on all nginx boxes after `nginx -s reload`.

server {
  listen 80;
  server_name drupal-sandbox.test;

  root /var/www/drupal/web;
  index index.php index.html;

  # Honor X-Forwarded-* from the ALB so Drupal sees real client IPs/scheme.
  # ALB is in 10.200.0.0/16 (project VPC); narrow this if you tighten subnets.
  set_real_ip_from 10.200.0.0/16;
  real_ip_header X-Forwarded-For;
  real_ip_recursive on;

  # Hidden files (allow well-known for ACME etc.)
  location ~ /\.(?!well-known) {
    deny all;
    log_not_found off;
    access_log off;
  }

  # Protect sensitive Drupal source files
  location ~* \.(engine|inc|install|module|profile|po|sh|.*sql|theme|twig|tpl(\.php)?|xtmpl|yml)(~|\.sw[op]|\.bak|\.orig|\.save)?$|^(\.(?!well-known).*|Entries.*|Repository|Root|Tag|Template|composer\.(json|lock)|web\.config)$|^#.*#$|\.php(~|\.sw[op]|\.bak|\.orig|\.save)$ {
    deny all;
    return 404;
  }

  # Image styles — generated on demand by Drupal
  location ~ ^/sites/.*/files/styles/ {
    try_files $uri @rewrite;
  }

  # Private files served via Drupal's /system/files route
  location ~ ^(/[a-z\-]+)?/system/files/ {
    try_files $uri /index.php?$query_string;
  }

  # Static assets — cache aggressively
  location ~* \.(css|js|png|jpg|jpeg|gif|ico|svg|woff2?|ttf|otf|eot|webp)$ {
    expires max;
    access_log off;
    log_not_found off;
    try_files $uri @rewrite;
  }

  # Clean URLs
  location / {
    try_files $uri /index.php?$query_string;
  }

  location @rewrite {
    rewrite ^ /index.php;
  }

  # PHP via the FastCGI upstream that configure-nginx.sh generates from
  # /sandbox/nlb/endpoint at boot. Switch to `php74` if you want the 7.4 pool.
  location ~ '\.php$|^/update\.php' {
    fastcgi_split_path_info ^(.+?\.php)(|/.*)$;
    try_files $fastcgi_script_name =404;

    include fastcgi_params;
    fastcgi_param SCRIPT_FILENAME $document_root$fastcgi_script_name;
    fastcgi_param PATH_INFO $fastcgi_path_info;
    fastcgi_param QUERY_STRING $query_string;
    fastcgi_param HTTP_PROXY "";

    fastcgi_pass php83;
    fastcgi_intercept_errors on;
    fastcgi_read_timeout 120s;
    fastcgi_buffers 16 16k;
    fastcgi_buffer_size 32k;
  }
}
NGXEOF
```
