**1. Why image-builder rolled back** (the actual error):

```bash
aws cloudformation describe-stack-events \
  --stack-name cf-scalable-web-sandbox-image-builder \
  --query 'StackEvents[?ResourceStatus==`UPDATE_FAILED` || ResourceStatus==`UPDATE_ROLLBACK_IN_PROGRESS`].[Timestamp,LogicalResourceId,ResourceStatus,ResourceStatusReason]' \
  --output table | head -40
```

**2. Whether the live PHP instances have our new code** (the smoking gun):

```bash
# Find a php83 instance and SSM-exec into it
INSTANCE=$(aws ec2 describe-instances \
  --filters "Name=tag:aws:autoscaling:groupName,Values=sandbox-php83-asg" \
            "Name=instance-state-name,Values=running" \
  --query 'Reservations[0].Instances[0].InstanceId' --output text)
echo "Probing $INSTANCE..."

aws ssm send-command --instance-ids "$INSTANCE" \
  --document-name "AWS-RunShellScript" \
  --parameters 'commands=["echo === configure-php.sh size ===","wc -l /opt/worxco/configure-php.sh","echo === has Drupal env block? ===","grep -c \"Drupal env vars\" /opt/worxco/configure-php.sh","echo === pool.d/extra contents ===","ls -la /etc/php/8.3/fpm/pool.d/extra/ 2>/dev/null || echo NO_EXTRA_DIR","echo === www.conf has include? ===","grep -c \"include=.*pool.d/extra\" /etc/php/8.3/fpm/pool.d/www.conf"]' \
```
